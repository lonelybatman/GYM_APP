Algorithmus — Set-Liste generierenfunction generateSetList(config: {  defaultSets: number,  isOneHanded: boolean,  hasDropSet: boolean,  hasBackoff: boolean,  warmupSets: number,   // aus letzter Session}): SetRow[] {  const rows: SetRow[] = [];  // Warmup-Sets (nur wenn letztes Mal getrackt)  for (let i = 1; i &lt;= config.warmupSets; i++) {    rows.push({ set_type: `W${i}` });  }  // Normal / Einhand Sets  for (let i = 1; i &lt;= config.defaultSets; i++) {    if (config.isOneHanded) {      rows.push({ set_type: `${i}L`, hand: &apos;L&apos; });      rows.push({ set_type: `${i}R`, hand: &apos;R&apos; });    } else {      rows.push({ set_type: String(i) });    }    // Drop-set direkt nach jedem Normal-Set    if (config.hasDropSet) {      rows.push({ set_type: &apos;D&apos;, parent_set_type: String(i) });    }  }  // Backoff am Ende  if (config.hasBackoff) {    rows.push({ set_type: &apos;B&apos; });  }  return rows;}
